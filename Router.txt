-------- konfiguracja adresu IP na  danym porcie -------------
enable
    config t
        int <port>
            ip address <address> <mask>
            no shutdown
            end


------- ustawiamy max wielkość ramki (po stronie na którą wysyłamy) -----
enable
    config t
        int <port>
            mtu 500

show ip interface brief     : opis wszystkich portów
show arp                    : tablica apr (IP -> Mac)
show ip route               : tablica routingu (którędy do danych IP)
no shutdown                 : włączamy port (interface), wymagane dla routerów
ip address <address> <mask> : ustawiamy adres IP dla danego portu w danej podsieci
mtu <size>                  : max wielkość ramki po stronie wysyłanej (my wysyłamy)
