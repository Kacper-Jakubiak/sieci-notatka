# ----- konfiguracja adresu IP na  danym porcie ------
enable
    config t
        int <port>
            ip address <address> <mask>
            no shutdown
            end


# ----------- ustawiamy max wielkość ramki -----------
// po stronie gdzie wysyłamy
enable
    config t
        int <port>
            mtu <size>

# ---------------- konfiguracja dhcp -----------------
enable
    config t
        service dhcp
        ip dhcp pool <name>
            network <ip> <mask>
            default-router <ip>
            dns-server <ip>
            exit
        ip dhcp excluded-address <start> <end>
        end

# ------------ konfiguracja serial portu -------------
enable
    conf t
        int <port>
            ip address <ip> <mask>
            clock rate <nr_rate> //TYLKO DCE
            no shutdown

# ----------------- trasa statyczna ------------------
enable
    conf t
        ip route <dokąd> <mask> <którędy>
                //siec  //maska //ip następnego routera

show ip interface brief     : opis wszystkich portów
show arp                    : tablica apr (IP -> Mac)
show ip route               : tablica routingu (którędy do danych IP)
no shutdown                 : włączamy port (interface), wymagane dla routerów
ip address <address> <mask> : ustawiamy adres IP dla danego portu w danej podsieci
mtu <size>                  : max wielkość ramki po stronie wysyłanej (my wysyłamy)
